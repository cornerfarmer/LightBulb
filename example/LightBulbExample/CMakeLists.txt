cmake_minimum_required(VERSION 2.8.9)
project (LightBulbExample)

file(GLOB_RECURSE SOURCES *.cpp *.hpp)
include_directories(.)

target_link_libraries(LightBulbExample ${wxWidgets_LIBRARIES})

target_link_libraries(LightBulbExample LightBulbApp)
